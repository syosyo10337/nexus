`docker-compose.yml`にて

```Ruby
tty: true

```

を追記すること

```Docker
e.g.)

version: '3.8'
services:
  paddock:
    container_name: kikan-paddock-as-zto-services
    build:
      context: .
      args:
        APP_MOUNT_PATH: ''
        RAILS_ENV:      'development'
    environment:
      APP_MOUNT_PATH:                 '/paddock'                      # URLのプレフィックス
      DB_HOST:                        'db'                            # データベースホスト
      DB_PORT:                        '3306'                          # データベースポート
      APP_KIKAN_URL:                  'http://interchange:3000'       # 基幹APIのURL
      UPGARAGE_URL:                   'https://www.dev.upgarage.com'  # 基幹APIのURL
      APP_KIKAN_CLIENT_ID:            'test'                          # 基幹APIのクライアントID
      APP_KIKAN_CLIENT_SECRET:        'test_secret'                   # 基幹APIのシークレットキー
      APP_MIGRATION_CREATE_TEST_DATA: 'true'                          # マイグレーション時にテスト用データを作成する場合は `true` (production環境では無効)
      APP_URL_HOST:                   'localhost:8084'                # アプリケーションが使用するホスト名
      APP_EMAIL_ADDRESS:              'test@upgarage.com'             # メールの送信元アドレス
      APP_URL_PROTOCOL:               'http'                          # アプリケーションが使用するプロトコル
      LEGACY_CB_PASSWORD_SECRET:      'himitsu'                       # Croooober会員のパスワード暗号化に使用する秘密鍵
      RAILS_WORKER:                   1                               # Railsのワーカー数
      SECRET_KEY_BASE:                'change_me'                     # Railsのシークレットキー
      KIKAN_ELASTICSEARCH_URL:        'http://elasticsearch:9200'     # Elasticsearch
    volumes:
      - .:/app
    ports:
      - '8084:3000'
		tty: true
		stdin_open: true
networks:
  default:
    name: docker.kikan2-local.internal
    external: true
```

## コンテナの標準入出力を有効にする

実際にデバッグを行う時には,

```Bash
$ docker compose up

# 別タブにて
# コンテナIDを確認して
$ docker container ls

$ docker container attach <container_id>
```

こうすることで、コンテナの標準入出力とローカルのターミナル上の標準入出力が繋がった状態になるので、debuggerやpryでステップ実行をするデバッガーを使うことができる。終了する際には`ctrl P + Q`
