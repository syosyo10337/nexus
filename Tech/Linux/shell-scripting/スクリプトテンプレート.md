# 実用的なスクリプトテンプレート

> **関連**: [シェルスクリプト.md](シェルスクリプト.md)｜[ベストプラクティス.md](ベストプラクティス.md)｜[引数とエスケープ.md](引数とエスケープ.md)（位置パラメータ・shift）

そのまま流用できる基本テンプレートと、オプション解析付きテンプレートをまとめます。

---

## 基本テンプレート

```bash
#!/usr/bin/env bash

set -euo pipefail

# スクリプトの説明
readonly SCRIPT_NAME="$(basename "$0")"
readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

# 使用方法を表示
usage() {
    cat <<EOF
Usage: $SCRIPT_NAME [OPTIONS] <argument>

Description:
    このスクリプトの説明

Options:
    -h, --help      このヘルプを表示
    -v, --verbose   詳細出力
    -d, --debug     デバッグモード

Examples:
    $SCRIPT_NAME file.txt
    $SCRIPT_NAME -v file.txt
EOF
}

# エラーで終了
die() {
    echo "ERROR: $*" >&2
    exit 1
}

# メイン処理
main() {
    # 引数チェック
    if [ $# -eq 0 ]; then
        usage
        exit 1
    fi
    
    # 処理
    echo "Processing..."
}

# スクリプト実行
main "$@"
```

## オプション解析付きテンプレート

```bash
#!/usr/bin/env bash

set -euo pipefail

# デフォルト値
VERBOSE=0
DEBUG=0

# オプション解析
while [[ $# -gt 0 ]]; do
    case $1 in
        -h|--help)
            usage
            exit 0
            ;;
        -v|--verbose)
            VERBOSE=1
            shift
            ;;
        -d|--debug)
            DEBUG=1
            set -x
            shift
            ;;
        -*)
            echo "Unknown option: $1" >&2
            usage
            exit 1
            ;;
        *)
            # 位置引数
            FILE="$1"
            shift
            ;;
    esac
done

# メイン処理
main() {
    [ "$VERBOSE" -eq 1 ] && echo "Verbose mode enabled"
    [ "$DEBUG" -eq 1 ] && echo "Debug mode enabled"
    
    # 処理
    echo "Processing: $FILE"
}

main
```
