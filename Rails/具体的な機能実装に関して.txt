def gravatar_for(user, size: 80)
	gravatar_id = Digest::MD5::hexdigest(user.email.downcase) 
	#gravtarでは,ユーザのメールアドレスをMD5でハッシュ化しているのでそれに対応させている。
	#上の通りにするとRubyでメールアドレスをMD5を使ったハッシュに変換できるんですよ。
	#MD5という形式では大文字小文字を区別してしまうので、.downcaseで変換させてます。これでDBからのユーザ画像用じゃないところから呼ばれる時にも対応できますne.
	gravatar_url = "https://secure.gravatar.com/avatar/#{gravatar_id}?s=#{size}"
	#メールアドレスをハッシュ化したものを末尾につけたURLアクセスすると画像が取得できる
	image_tag(gravatar_url, alt: user.name, class: "gravatar") 
end

メソッドを独自に定義して。sessionコントローラだけでなく他の場所でも使えるようにする。

module SessionsHelper
	
	# 渡されたユーザーでログインする
	def log_in(user)
		session[:user_id] = user.id
	end
	
	# 現在ログイン中のユーザーを返す（いる場合）
	def current_user#(A)
		if session[:user_id]
			@current_user ||= User.find_by(id: session[:user_id])
		end
	end
	
	# ユーザーがログインしていればtrue、その他ならfalseを返す
	def logged_in?
		!current_user.nil?
	end
end


#(A)の部分は、以下が原型で、or equalsの記法で簡略化している。

	@current_user

def User.digest(string)
	cost = ActiveModel::SecurePassword.min_cost ? BCrypt::Engine::MIN_COST :
	BCrypt::Engine.cost
	BCrypt::Password.create(string, cost: cost)
end

{step2-0}store_locationとredirect_back_or(default)をsessions_helperに用意する。
@「app/helpers/sessions_helper.rb」

#ユーザがリクエストしてたURLを記憶する
def store_location
	#ユーザがリクエストしたURL(=request.original_url)をsessionメソッドで記憶する。
	session[:forwarding_url] = request.original_url if request.get?#(A)
end

#記憶したURL(or default)にリダイレクト
def redirect_back_or(default)
	#if session[:] = nil -> redirect_to(user)となるように引数を取っている 
	redirect_to(session[:forwarding_url] || default)
	session.delete(:forwarding_url)
	#returnするか、式が最後まで評価された後に、redirectが発生するので、２行目のセッション削除は常に有効です。
end

**store_locationメソッドでは、リクエストが送られたURLをsession変数の:forwarding_urlキーに格納する。ただし、GETリクエストのときのみ格納するようにする(request.get?)#(A)
リダイレクト先として、POST(PATCH/DELETE)リクエストを期待しているURLが入っていることで、エラーが発生する可能性がある。
ex)ユーザがsession[:user_id]等のセッション実現のためのcookieを手動で削除して,ログイン状態を改変した状態で、フォーム送信すると、未ログイン状態だと認識され、beforeフィルターに引っかかり,store_locationを実行させてしまう.

{step3-1}activation_tokenと_digestの値が一致するか判定するメソッドを定義(Remember meで実装したauthenicated?メソッド流用して一般化)

def authenticated?(attribute, token)
	#モデル内にあるので、selfは省略可
	#	digest = (self).send("#{attribute}_digest")
	digest = send("#{attribute}_digest")
	return false if digest.nil?
	BCrypt::Password.new(digest).is_password?(token)
end

def edit
	user = User.find_by(email: params[:email])
	if user && !user.activated? && user.authenticated?(:activation, params[:id])
		#!user.activated?は、有効化されたユーザを再度有効しないため(第三者がリンクを踏むことでactivatedを上書きしてログインするのを避ける)
		#params[:id]には、tokenが入っている
	
		user.update_attribute(:activated, true)
		user.update_attribute(:activated_at, Time.zone.now)
		#update_attributesだとvalidationがかかるため、別々に記述している
	
		#users#create(->signup)の時にsessionをつけないで、メールを送るようにしたので、
		log_in user
		flash[:success] = "Your account activated!"
		redirect_to user
	else
		#もし何かの手違いで、条件が満たされない時(不正なアクセス等)
		flash[:danger] = "Invalid activation link"
		redirect_to root_url
	end
end

def create
	@user = User.find_by(email: params[:session][:email].downcase)
	if @user &.authenticate(params[:session][:password])
		if @user.activated?
			log_in @user
			params[:session][:remember_me] == '1' ? remember(@user) : forget(@user)
			redirect_back_or @user
		else
			message = "Account no activated. "
			message += "Check your email for the activation link"
			flash[:warning] = message
			redirect_to root_url
		end
	else
		flash.now[:danger] = "Invalid email/password combination"
		render "new"
	end

***[パスワード再設定機能(12章)]=================

大まかなFlow(画面遷移)------------
1) ユーザーから、パスワードリセットをリクエストされた際に、そのユーザから送信されたemailを元に探す。
2) そのemailがDBに存在する時、パスワードリセット用の_tokenとそれに対応する_digestを生成する。
3) _digestはDBで保存し、その後ユーザに対してメールをreset_tokenとuserのemailを含んだリンクを埋め込んで送信する。
4) userがリンクを踏んだ時に、emailを元にユーザを検索し、_tokenと_digestを比較する形で認証する。
5) 認証が済むと、ユーザにパスワード再設定フォーム(画面)を表示する。
------------------

パスワード再設定リソースを作成する。AccountActivation同様ActiveRecordとは関連付けずに、関連するデータはUserモデル自身に含める。リソースとして扱うことにしたので、RESTなURL経由でやり取りできるように実装を始める。
再設定リソースをcreateし,update(更新)すること、newとeditフォームをレンダーも行うことで、今回はRESTfulな4つのroutesを用意する。

def create
	@user = User.find_by(email: params[:password_reset][:email])
	if @user
		#token生成->digest保存と、timestamp更新するメソッド
		@user.create_reset_digest
		
		#userにメールを送信するメソッド
		@user.send_password_reset_email

		redirect_to root_url
	else
		flash[:danger] = "Email address no found"
		render "new"
	end
end

{step3-3}updateアクションを実装(パスワード書き換える)

考慮すべき項目

②無効なパスワードであれば失敗させる（失敗した理由も表示する）

.

def update
	if params[:user][:password].empty? #③の対策
		#エラーメッセージ追加するerrors.add(non-custom)
		#今回はpasswordがblankの時に、それ用のエラーメッセージが追加されるように
		#@userはbefore_filterで取得してきている。
		@user.errors.add(:password, :blank)
		render "edit"
	
		#適格な値が入力された時
	elsif @user.update(user_params)
		log_in @user
		#一度再設定が完了したら、再度メールアドレスを求めるようにする。
		@user.update_attribute(:reset_digest, nil)
		password[:success] = "Password has been reset"
		redirect_to @user
	else #②
		render "edit"
	end
end

private 
	#ユーザがフォームからデータをPOSTするような場合にはmass assignになることを避ける。
	def user_params
		params.require(:user).permit(:password, :password_confirmation)
	end
.
.
	def check_expiration #①への対策、再設定の有効期限をチェックする機能
		if @user.password_reset_expired?
			flash[:danger] = "Password reset has expired."
			redirect_to new_password_reset_url
		end
	end

def destroy
	@micropost.destroy

	redirect_to request.referrer || root_url

def correct_user
	#選択されている投稿がユーザ自身のものである時のみ変数を定義
	@micropost = current_user.microposts.find_by(id: params[:id])
	redirect_to root_url if @micropost.nil?
end

{step1-4}多対他のデータ関連付け
ここでの中間テーブル(joinモデル)は、relationshipsですね

デフォルトのhas_many throughという関連付けでは、Railsはモデル名（単数形）に対応する外部キーを探すので、
has_many :followeds, through: :active_relationships
#active_relationships(つまりRelationshipモデルの中の外部キー(カラム)の中から、followedとなっているものを探し、その外部キーを元に参照先の集合を取得する。)
ただ、followedsという名前が気に食わないので、
:sourceパラメーターを使って、「following配列の元は:followedによるuserデータの集合である」と認識させる。


これにより、フォローしているユーザーを配列の様に扱えるようになりました。
user.following.include?(other_user)
user.following.find(other_user)が使えますね！
user.following << other_user
user.following.delete(other_user)で、配列のようにuser.followingで取得したオブジェクトを操作できる。

#following.include?(other_user)の注意点
フォローしている全てのユーザーをデータベースから取得した後に、その配列に対してinclude?メソッドを実行しているように見えますが、
実際にはデータベースの中で直接比較をするように動いている。クエリ見てみろ。

作成したmacroメソッドを元にしてCRUD操作を可能にする
follow/unfollow/following?などのメソッドを定義する。

#CGI.espaceHTMLを使ってmicropost.contentをエスケープさせる理由

まず、
①response.bodyは、<body></body>だけでなく、HTML全文を示している。
テストの文脈で使う時は、今レンダーされるHTMLのコンテンツ全てを対象にして、特定の文字列があるか？というように使われる。

(②)Rubyにおいて(''文字列リテラルと見做される)などは特別な意味をもつ特殊文字になるので、response.bodyで表現されるHTML全文の内容はエスケープされている。
③HTMLのコンテンツに当たる部分は、HTMLエスケープが施されている
例えば、<a>I'm sorry.、</a>があるとすれば、
エスケープ処理をしてI&#39;m sorry.と表記されているはずである。

この時,'HTMLのコンテンツ(タグの内部)に当たる部分がHTMLエスケープつまり,
'&' => '&amp;',
'<' => '&lt;',
'>' => '&gt;',
'"' => '&quot;',
"'" => '&#39;'で表現されているresponse.bodyを対象にして、
matchする表現を探すため、検索条件(ここでは、.contentメソッドで取得してきた。投稿内容の素の文字列)もエスケープさせる。
