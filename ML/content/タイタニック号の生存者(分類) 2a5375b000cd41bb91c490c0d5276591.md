# 前処理

- 欠損値について、あまりにデータの総数に近く欠損している場合には、まずは次元としてカウントしないこととする。

- その他の要素については、平均値や中央値、最頻値で埋める。

- 決定木においては、外れ値の影響を受けづらいので、確認して排除する必要はない。

### 一連の流れ

```Python
import pandas as pd
from sklearn import tree
from sklearn.model_selection import train_test_split
import pickle

df = pd.read_csv('./datafiles/Survived.csv')
df.head(2)


#  欠損値を確認する
print(df.isnull().any(axis=0))
df.isnull().sum()
# 全体のデータするのうちどれくらいの割合で欠損値があるかを確認する。
df.shape

# 欠損値補完
# Cabinについては欠損値がentire datasetの値にかなり近かったので取り除く
# Ageを平均値で Embarkedを最頻値で埋めてみる
df['Age'] = df['Age'].fillna(df['Age'].mean())
df['Embarked'] = df['Embarked'].fillna(df['Embarked'].mode()[0])



# 試しに説明変数と目的変数に分割する(特徴量には数値だけが取れるのでEmbarkedも無視)
x_col = ['Pclass', 'Age', 'SibSp', 'Parch', 'Fare']

x = df[x_col]
y = df['Survived']

x_train, x_test, y_train, y_test = train_test_split(x, y, test_size= 0.2, random_state = 0)
# 一応適切に分割できているか確認
x_train.shape



# モデルの作成と学習
model = tree.DecisionTreeClassifier(max_depth=5,
                                    random_state=0,
                                    class_weight = 'balanced')

# class_weight = 'balanced'の指定をすることで、正解データの割合が不均衡の場合でも
# 分岐条件を考える際に、比率の大きいデータの影響を小さく、小さいデータの影響をおおきくし、分岐条件の決定をbalancedさせることで、予測性能を向上させる効果が期待できる。
model.fit(x_train, y_train)
model.score(x_test, y_test)
#-> 0.7374301675977654



# モデルのチューニングが必要だと分かったので、学習と評価する過程を関数化
# x,yと木の深さを引数に取り、train_score/test_score, とモデル自身を返す
def learn(x, y, depth=3):
  x_train, x_test, y_train, y_test = train_test_split(x,
                                                      y,
                                                      test_size = 0.2,
                                                      random_state = 0)
  
  model = tree.DecisionTreeClassifier(max_depth = depth,
                                      random_state = 0,
                                      class_weight = 'balanced')
  model.fit(x_train, y_train)
  
  train_score = model.score(X = x_train, y = y_train)
  test_score = model.score(X = x_test, y = y_test)
  
  return round(train_score, 3), round(test_score, 3), model


for j in range(1, 15):
  train_score, test_score, model = learn(x, y, depth = j)
  print(f'''
木の深さ{j}の時
訓練データの正解率: {train_score}
テストデータの正解率: {test_score}
        ''')

# データの前処理を考察する

# 1).欠損値の補完を中央値にする。
df2 = pd.read_csv('./datafiles/Survived.csv')
print(df2['Age'].mean())
print(df2['Age'].median())
#-> 大差なし

# 2) 生存したか否かの平均年齢
# 3)　乗客のクラス別の平均年齢を見てみる
df2.groupby('Survived').mean(numeric_only=True)['Age']
df2.groupby('Pclass').mean(numeric_only=True)['Age']

# 4)試しに、乗客クラスxisSurvivedの二軸ごとの平均年齢を取る
# ピボットテーブル機能(pandas)を用いて、クロス集計を行う。
pd.pivot_table(df2, index= 'Survived',columns = 'Pclass', values = 'Age')

#->
Pclass	1	2	3
Survived			
0	43.695312	33.544444	26.555556
1	35.368197	25.901566	20.646118



# PlassとSurvivedの区分ごとにAgeの欠損値を補完する
is_null = df2['Age'].isnull()

df2.loc[(df2['Pclass'] == 1) \
        & (df2['Survived'] == 0) \
        & (is_null), 'Age'] \
= 43
df2.loc[(df2['Pclass'] == 1) \
        & (df2['Survived'] == 1) \
        & (is_null), 'Age'] \
= 35

df2.loc[(df2['Pclass'] == 2) \
        & (df2['Survived'] == 0) \
        & (is_null), 'Age'] \
= 33
df2.loc[(df2['Pclass'] == 2) \
        & (df2['Survived'] == 1) \
        & (is_null), 'Age'] \
= 25

df2.loc[(df2['Pclass'] == 3) \
        & (df2['Survived'] == 0) \
        & (is_null), 'Age'] \
= 26
df2.loc[(df2['Pclass'] == 3) \
        & (df2['Survived'] == 1) \
        & (is_null), 'Age'] \
= 20





# 新しく前処理を施したデータを元にモデルを使ってみる。
x_col = ['Pclass', 'Age', 'SibSp', 'Parch', 'Fare']
x = df2[x_col]
y = df2['Survived']

for j in range(1, 15):
  train_score, test_score, model = learn(x, y, depth = j)
  print(f'''
木の深さ{j}の時
訓練データの正解率: {train_score}
テストデータの正解率: {test_score}
        ''')

#->まだ0.8超えません

# 5)性別によって生存率に違いがある？
# plotでの図示で確認
grouped_by_sex = df2.groupby('Sex').mean(numeric_only=True)

# index(グループ分けされた)sexを横軸
# survivedカラムの値の平均値を縦軸とした棒グラフを図示
grouped_by_sex['Survived'].plot(kind = 'bar')




# 実際に特徴量にsexを追加して学習させてみる。

male_dummies = pd.get_dummies(df2['Sex'], drop_first = True)
new_x = pd.concat([x, male_dummies], axis = 1)

for j in range(1, 15):
  train_score, test_score, model = learn(new_x, y, depth = j)
  print(f'''
木の深さ{j}の時
訓練データの正解率: {train_score}
テストデータの正解率: {test_score}
        ''')
## depth =9 がめちゃ良さそう。
*scores, model = learn(new_x, y, 9)

with open('./machine_learning/ch7/survived.pkl', 'wb') as f:
  pickle.dump(model, f)


# 特徴量重要度を評価する
fi = model.feature_importances_
# dfにして見やすく表示
pd.DataFrame(fi, index = new_x.columns)
```
