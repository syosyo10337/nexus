```Python
# モジュールのインストール
import pandas as pd
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split
## データの標準化を行うためのモジュール
from sklearn.preprocessing import StandardScaler
import pickle

# データの読み込み
df = pd.read_csv('./datafiles/Boston.csv')
df.head(2)


# ------------------------------------------------
# 文字列の列をダミー変数化
## (不要になった'CRIME'を削除も忘れず)
df['CRIME'].value_counts()
crime = pd.get_dummies(df['CRIME'], drop_first= True)
df2 = pd.concat([df,crime], axis=1)
df2 = df2.drop('CRIME', axis=1)

df2.head(2)


# 前処理をする前にdatasetsの分割
# ->train + vaidation & test
train_val, test = train_test_split(
  df2, 
  test_size = 0.2,
  random_state = 0
)


# 欠損値確認と補完(train/val)
train_val.isnull().sum()
train_val_mean = train_val.mean()
train_val2 = train_val.fillna(train_val_mean)
train_val2.isnull().sum() #確認



# 外れ値の処理
## 散布図を描画
cols = train_val2.columns
for col in cols:
  train_val2.plot(kind = 'scatter', x = col, y = 'PRICE')

##　極端な外れ値を削除
RM_outlier_index = train_val2[(train_val2['RM'] < 6) & \
	(train_val2['PRICE'] > 40)].index
PTRATIO_outlier_index = train_val2[(train_val2['PTRATIO'] > 20 ) & \ 
	(train_val2['PRICE'] > 40)].index

## たまたまどちらのも[76]のデータだったので
train_val3 = train_val2.drop([76], axis= 0)

## 特徴量を絞り込む(散布図を参考に、相関があるものを抽出)
train_val3.columns
col = [
  'INDUS',
  'NOX',
  'RM',
  'PTRATIO',
  'LSTAT',
  'PRICE'
]

train_val4 = train_val3[col]
train_val4.head(3)


# 相関係数を調べてさらに絞り込み
# 調べたい相関についての相関行列だけ取得する(PRICE :x -> ?)
train_val_corr_with_price = train_val4.corr()['PRICE']
train_corr

# se.map処理を活用して、絶対値を省いた相関を比較
train_corr.map(abs).sort_values(ascending　=　False)

# -> 特徴量を RM / LSTAT / PTRATIOにすること決めた。
feature_cols = [
  'RM',
  'LSTAT',
  'PTRATIO'
]
x = train_val4[feature_cols]
y = train_val4[['PRICE']] # 標準化のためにdfで作成

# train_val -> train + valへ
x_train, x_vali, y_train, y_vali = train_test_split(
  x,
  y,
  test_size = 0.2,
  random_state = 0
)



# データの標準化
## StandardScalerクラスのオブジェクトを生成する
sc_model_x = StandardScaler()

## fitメソッドにデータフレームを渡すと、
## そのデータフレームの平均値と標準偏差を属性として保持したオブジェクトになる
sc_model_x.fit(x_train)

## transformメソッドにデータセットを渡すとき
## オブジェクトが持つ平均値と標準偏差を元に、標準化されたデータを返す。
sc_x = sc_model_x.transform(x_train)
sc_x

## データの標準化ができているかの確認
## -16乗ってことはほぼ０/stdほぼほぼ1
tmp_df = pd.DataFrame(sc_x, columns = x_train.columns)
print(tmp_df.mean())
print(tmp_df.std())

# 正解データに関しても標準化を行う
sc_model_y = StandardScaler()
sc_model_y.fit(y_train)
sc_y = sc_model_y.transform(y_train)
sc_y



# ------------------------------------------------
# [回帰分析のモデルの作成と学習]
model = LinearRegression()
model.fit(sc_x, sc_y)

# validation_dataで性能を検証
sc_x_vali = sc_model_x.transform(x_vali)
sc_y_vali = sc_model_y.transform(y_vali)

model.score(sc_x_vali, sc_y_vali) # 標準化データを用いる

# チューニング用に関数を用意
def learn(x,y):
  x_train, x_vali, y_train, y_vali = train_test_split(
		x, y,
		random_state = 0,
    test_size = 0.2
	)

  sc_x_model = StandardScaler()
  sc_y_model = StandardScaler()
  sc_x_model.fit(x_train)
  sc_y_model.fit(y_train)
  
  sc_x_train = sc_x_model.transform(x_train)
  sc_y_train = sc_y_model.transform(y_train)
  
  model = LinearRegression()
  model.fit(sc_x_train, sc_y_train)
  
  sc_x_vali = sc_x_model.transform(x_vali)
  sc_y_vali = sc_y_model.transform(y_vali)
  
  train_score = model.score(sc_x_train, sc_y_train)
  vali_score = model.score(sc_x_vali, sc_y_vali)
  
  return train_score, vali_score

# 関数をテスト
x = train_val4.loc[ : , ['RM', 'LSTAT', 'PTRATIO']]
y = train_val4[['PRICE']]

ts,vs = learn(x,y)
print(ts,vs)
# ->
0.717589757251598 0.7359028880291

# チューニングをしていく
# 多項式特徴量を追加する
x['RM2'] = x['RM'] ** 2
x.head(2)
# 評価
train_score, vali_score = learn(x, y)
print(train_score, vali_score)

# 試しに他の２乗も足してみるか
col = ['LSTAT', 'PTRATIO']
for i in col:
  x[f'{i}2'] =x[i] ** 2
  print(*learn(x, y))

# 交差作用特徴量も追加してみる
x['RM * LSTAT'] = x['RM'] * x['LSTAT']
x.head(2)

print(*learn(x, y))
# validationデータでのスコアが0.87を超えました
# ここでチューニングを終了とします。



# ------------------------------------------------
# テストの前にtraining + validationのデータを用いて再学習 
#  データの標準化を行う
# xは必要な特徴量datasets/ yは'PRICE'datasets

sc_model_x2 = StandardScaler()
sc_model_y2 = StandardScaler()
sc_model_x2.fit(x)
sc_model_y2.fit(y)
sc_x2 = sc_model_x2.transform(x)
sc_y2 = sc_model_y2.transform(y)

model = LinearRegression()
model.fit(sc_x2, sc_y2)


# 手付かずだったtest_dataの前処理を行う
# 1. 欠損値をtrain_valの平均値で埋める
# 2. 外れ値を削除しない(未知のデータという意味合いでのテストなので削除しない)
# 3. (特徴量と正解データを決める)確認する
# 4. 多項特徴量 交差作用特徴量も同様に追加する
# 5. データを標準化する。

test2 = test.fillna(train_val_mean)
test_x = test2.loc[ : , ['RM', "LSTAT", "PTRATIO"]]
test_y = test2[['PRICE']]

cols = [
  'RM',
  "LSTAT",
  "PTRATIO"
]

for col in cols:
  test_x[f'{col}2'] = test_x[col] ** 2

test_x['RM * LSTAT'] = test_x['RM'] * test_x['LSTAT']

test_sc_x = sc_model_x2.transform(test_x)
test_sc_y = sc_model_y2.transform(test_y)

model.score(test_sc_x, test_sc_y)
# -> 0.7649249353669053
```
