## 一連の流れ

```Python
# 必要なモジュールのimport
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_absolute_error
import pickle


# データの読み込み
df = pd.read_csv('./datafiles/cinema.csv')
df.head(3)

# 欠損値の確認と、補完
df.isnull().any(axis=0)

df2 = df.fillna(df.mean())
df2.isnull().any(axis=0)


# 外れ値を散布図を描画し、確認
%matplotlib inline

for column in df2.columns[1:5]:
  df2.plot(kind = 'scatter', x = column, y = 'sales')


# 調べた外れ値を削除する
outlier_index = df2[(df2['SNS2'] > 1000) & (df2['sales'] < 8500)].index
df3 = df2.drop(outlier_index, axis=0)
df3.shape # 行数で削除できているかを確認


# 特徴量と正解データ(説明変数と目的変数)を用意して
x = df3.loc[ : , "SNS1":'original']
y = df3['sales']

x_train, x_test, y_train, y_test = train_test_split(
  x, y, test_size = 0.2, random_state = 0
)

# modelの作成と学習
model = LinearRegression()
model.fit(x_train, y_train)


# 予測させてみる。
new = pd.DataFrame([[150, 700, 300, 0]], columns= x_train.columns)
model.predict(new)

# MAEを算出
pred = model.predict(x_test)
mean_absolute_error(y_pred = pred, y_true = y_test)

# 決定係数を算出
model.score(x_test, y_test)

# モデルを保存
with open('./machine_learning/ch6/cinema.pkl', 'wb') as f:
  pickle.dump(model, f)



# (おまけ)回帰式を調べる。
print(model.coef_)
print(model.intercept_)

tmp = pd.DataFrame(model.coef_)
tmp.index = x_train.columns
tmp
```
